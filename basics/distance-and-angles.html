Explanation

In CustomPainter you often need the distance between two points (e.g. to scale or clamp sizes) and the angle from one point to another (e.g. to orient a shape or compute direction). Both come from the same idea: treat the segment between two points as a vector.

The distance formula gives the length of that segment. The atan2 function gives the angle (in radians) from the positive x-axis to the segment, which is exactly what you need for direction, rotation, or placing items along a line. In Flutter, Offset provides Offset.zero.directionTo(point) and distance helpers; for raw math you use the distance formula and dart:math‚Äôs atan2.

Use cases: hit testing (distance to a point), pointing shapes toward a target (angle), dashed lines (distance along path), and any layout that depends on ‚Äúhow far‚Äù or ‚Äúwhich direction.‚Äù

Formulas

Distance formula

Distance between two points (x‚ÇÅ, y‚ÇÅ) and (x‚ÇÇ, y‚ÇÇ):

distance = ‚àö((x‚ÇÇ ‚àí x‚ÇÅ)¬≤ + (y‚ÇÇ ‚àí y‚ÇÅ)¬≤)

In Dart you can use math.sqrt(dx*dx + dy*dy) where dx = p2.dx - p1.dx, dy = p2.dy - p1.dy, or use (p2 - p1).distance with Offset.

atan2 angle formula

Angle (in radians) from the positive x-axis to the vector (dx, dy):

angle = atan2(dy, dx)

Result is in the range (‚àíœÄ, œÄ]. In Flutter, Offset has a direction getter that returns this angle. Use it for rotation, direction to a target, or converting back to unit vector with cos/sin.

Visual Demo

class DistanceAnglesDemoScreen extends StatefulWidget {
  const DistanceAnglesDemoScreen({super.key});

  @override
  State<DistanceAnglesDemoScreen> createState() => _DistanceAnglesDemoScreenState();
}

class _DistanceAnglesDemoScreenState extends State<DistanceAnglesDemoScreen> {
  late Offset _p1;
  late Offset _p2;
  Offset? _dragging;

  @override
  void initState() {
    super.initState();
    _p1 = const Offset(80, 200);
    _p2 = const Offset(280, 320);
  }

  void _onPanStart(DragStartDetails d, Size size) {
    final pos = d.localPosition;
    const hitRadius = 24.0;
    final d1 = sqrt((pos.dx - _p1.dx) * (pos.dx - _p1.dx) + (pos.dy - _p1.dy) * (pos.dy - _p1.dy));
    final d2 = sqrt((pos.dx - _p2.dx) * (pos.dx - _p2.dx) + (pos.dy - _p2.dy) * (pos.dy - _p2.dy));
    if (d1 < hitRadius) { _dragging = _p1; }
    else if (d2 < hitRadius) { _dragging = _p2; }
  }

  void _onPanUpdate(DragUpdateDetails d) {
    if (_dragging == null) return;
    setState(() {
      if (_dragging == _p1) { _p1 = Offset(_p1.dx + d.delta.dx, _p1.dy + d.delta.dy); }
      else { _p2 = Offset(_p2.dx + d.delta.dx, _p2.dy + d.delta.dy); }
    });
  }

  void _onPanEnd(DragEndDetails _) { setState(() => _dragging = null); }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Distance & Angles Demo'), centerTitle: true),
      body: LayoutBuilder(
        builder: (context, constraints) {
          return GestureDetector(
            onPanStart: (d) => _onPanStart(d, constraints.biggest),
            onPanUpdate: _onPanUpdate,
            onPanEnd: _onPanEnd,
            child: CustomPaint(
              painter: DistanceAnglesDemoPainter(p1: _p1, p2: _p2),
              size: constraints.biggest,
            ),
          );
        },
      ),
    );
  }
}

class DistanceAnglesDemoPainter extends CustomPainter {
  DistanceAnglesDemoPainter({required this.p1, required this.p2});
  final Offset p1;
  final Offset p2;

  @override
  void paint(Canvas canvas, Size size) {
    // ---- Line connecting the two points ----
    final linePaint = Paint()
      ..color = Colors.blue
      ..strokeWidth = 3
      ..style = PaintingStyle.stroke;
    canvas.drawLine(p1, p2, linePaint);

    // ---- Distance: sqrt((x2-x1)¬≤ + (y2-y1)¬≤) ----
    final dx = p2.dx - p1.dx;
    final dy = p2.dy - p1.dy;
    final distance = sqrt(dx * dx + dy * dy);

    // ---- Angle (radians): atan2(y2-y1, x2-x1) ‚Äî then convert to degrees ----
    final angleRad = atan2(dy, dx);
    final angleDeg = angleRad * 180 / 3.14159265359;

    // ---- Draw the two points ----
    final pointPaint = Paint()..color = Colors.orange..style = PaintingStyle.fill;
    canvas.drawCircle(p1, 16, pointPaint);
    canvas.drawCircle(p2, 16, pointPaint);
    final strokePaint = Paint()..color = Colors.deepOrange..style = PaintingStyle.stroke..strokeWidth = 2;
    canvas.drawCircle(p1, 16, strokePaint);
    canvas.drawCircle(p2, 16, strokePaint);

    // ---- Display distance and angle on screen ----
    final textStyle = TextStyle(color: Colors.grey[800], fontSize: 18, fontWeight: FontWeight.w600);
    final paragraph = TextPainter(
      text: TextSpan(style: textStyle, children: [
        const TextSpan(text: 'Distance = '),
        TextSpan(text: distance.toStringAsFixed(1), style: TextStyle(color: Colors.blue[700])),
        const TextSpan(text: '\nAngle = '),
        TextSpan(text: '${angleDeg.toStringAsFixed(1)}¬∞', style: TextStyle(color: Colors.blue[700])),
        const TextSpan(text: ' (atan2(Œîy, Œîx))'),
      ]),
      textDirection: TextDirection.ltr,
    )..layout(maxWidth: size.width - 32);
    paragraph.paint(canvas, const Offset(16, 24));
  }

  @override
  bool shouldRepaint(covariant DistanceAnglesDemoPainter oldDelegate) {
    return oldDelegate.p1 != p1 || oldDelegate.p2 != p2;
  }
}
