üìö What is Offset?

Offset is a class in Flutter (from dart:ui) that represents a 2D point or position in the coordinate system. It's essentially a pair of X and Y coordinates that tells you WHERE to draw something on the canvas.

Think of Offset as:

  • A point on a graph (x, y coordinates)
  • A location on the canvas
  • A position vector from the origin (0, 0)
  • The "address" of where to place shapes, lines, and text
üí° Key Concept: Offset is used extensively in CustomPainter's paint() method to specify positions. Almost every Canvas drawing method requires an Offset to know where to draw. For example, canvas.drawCircle(Offset(100, 50), 30, paint) draws a circle centered at position (100, 50).

Structure

class Offset {
  final double dx; // Horizontal position (X coordinate)
  final double dy; // Vertical position (Y coordinate)
}

üéØ Essential Offset Properties & Methods

Properties: dx and dy

Type: double dx, double dy

Definition: The horizontal (dx) and vertical (dy) components of the offset.

What They Represent:

  • dx - Distance from the left edge (X-axis position)
  • dy - Distance from the top edge (Y-axis position)
  • Both are measured in logical pixels
  • Can be positive, negative, or zero
@override
void paint(Canvas canvas, Size size) {
  final point = Offset(100, 50);

  // Access individual components
  print('X position: ${point.dx}'); // 100.0
  print('Y position: ${point.dy}'); // 50.0

  // Use in drawing
  canvas.drawCircle(point, 30, Paint()..color = Colors.blue);

  // Create offset from size (center of canvas)
  final center = Offset(size.width / 2, size.height / 2);
  canvas.drawCircle(center, 40, Paint()..color = Colors.red);
}

Constructor: Offset(dx, dy)

Purpose: Creates an Offset with specific X and Y coordinates.

@override
void paint(Canvas canvas, Size size) {
  // Create offset at position (100, 50)
  final point1 = Offset(100, 50);

  // Create offset at top-left corner
  final topLeft = Offset(0, 0);

  // Create offset at bottom-right corner
  final bottomRight = Offset(size.width, size.height);

  // Draw circles at these positions
  canvas.drawCircle(point1, 20, Paint()..color = Colors.blue);
  canvas.drawCircle(topLeft, 15, Paint()..color = Colors.red);
  canvas.drawCircle(bottomRight, 15, Paint()..color = Colors.green);
}

Static: Offset.zero

Purpose: Creates an Offset at the origin (0, 0) - the top-left corner.

@override
void paint(Canvas canvas, Size size) {
  // Offset.zero is the same as Offset(0, 0)
  final origin = Offset.zero;

  // Draw at top-left corner
  canvas.drawCircle(origin, 25, Paint()..color = Colors.blue);

  // Common pattern: Create rect from origin
  final rect = origin & Size(100, 50);
  canvas.drawRect(rect, Paint()..color = Colors.red);
}

Method: distance

Type: double distance

Purpose: Calculates the Euclidean distance from the origin (0, 0) to this offset.

Formula: ‚àö(dx¬≤ + dy¬≤)

@override
void paint(Canvas canvas, Size size) {
  final point = Offset(30, 40);
  final dist = point.distance; // ‚àö(30¬≤ + 40¬≤) = 50.0

  // Use distance to check if point is within a circle
  final center = Offset(size.width / 2, size.height / 2);
  final radius = 50.0;

  if (point.distance < radius) {
    // Point is inside the circle
    canvas.drawCircle(point, 10, Paint()..color = Colors.green);
  } else {
    canvas.drawCircle(point, 10, Paint()..color = Colors.red);
  }
}

Method: distanceTo(Offset other)

Type: double distanceTo(Offset other)

Purpose: Calculates the distance between two offsets.

@override
void paint(Canvas canvas, Size size) {
  final point1 = Offset(50, 50);
  final point2 = Offset(150, 100);

  // Calculate distance between two points
  final dist = point1.distanceTo(point2);

  // Draw a line connecting the points
  canvas.drawLine(
    point1,
    point2,
    Paint()
      ..color = Colors.blue
      ..strokeWidth = 2,
  );

  // Draw circles at both points
  canvas.drawCircle(point1, 8, Paint()..color = Colors.red);
  canvas.drawCircle(point2, 8, Paint()..color = Colors.green);

  // Display distance (if using TextPainter)
  print('Distance: $dist');
}

Method: scale(double scaleX, double scaleY)

Type: Offset scale(double scaleX, double scaleY)

Purpose: Returns a new Offset with scaled coordinates.

@override
void paint(Canvas canvas, Size size) {
  final original = Offset(50, 30);

  // Scale by 2x in both directions
  final scaled = original.scale(2.0, 2.0); // Offset(100, 60)

  // Scale only horizontally
  final horizontalScaled = original.scale(2.0, 1.0); // Offset(100, 30)

  // Draw original and scaled points
  canvas.drawCircle(original, 5, Paint()..color = Colors.blue);
  canvas.drawCircle(scaled, 5, Paint()..color = Colors.red);
  canvas.drawCircle(horizontalScaled, 5, Paint()..color = Colors.green);
}

Method: translate(double translateX, double translateY)

Type: Offset translate(double translateX, double translateY)

Purpose: Returns a new Offset moved by the given amounts.

@override
void paint(Canvas canvas, Size size) {
  final start = Offset(50, 50);

  // Move 100 pixels to the right, 30 pixels down
  final moved = start.translate(100, 30); // Offset(150, 80)

  // Move only horizontally
  final right = start.translate(50, 0); // Offset(100, 50)

  // Draw a path showing the movement
  final path = Path()
    ..moveTo(start.dx, start.dy)
    ..lineTo(moved.dx, moved.dy);

  canvas.drawPath(
    path,
    Paint()
      ..color = Colors.blue
      ..strokeWidth = 2,
  );

  canvas.drawCircle(start, 8, Paint()..color = Colors.red);
  canvas.drawCircle(moved, 8, Paint()..color = Colors.green);
}

Operators: +, -, *, /

Purpose: Perform arithmetic operations on offsets.

@override
void paint(Canvas canvas, Size size) {
  final point1 = Offset(50, 30);
  final point2 = Offset(20, 10);

  // Addition: Combine two offsets
  final sum = point1 + point2; // Offset(70, 40)

  // Subtraction: Difference between offsets
  final diff = point1 - point2; // Offset(30, 20)

  // Multiplication: Scale by a number
  final scaled = point1 * 2.0; // Offset(100, 60)

  // Division: Divide by a number
  final divided = point1 / 2.0; // Offset(25, 15)

  // Practical example: Calculate midpoint
  final midpoint = (point1 + point2) / 2.0; // Offset(35, 20)

  canvas.drawCircle(midpoint, 10, Paint()..color = Colors.blue);
}

Common Patterns in CustomPainter

Finding the Center

@override
void paint(Canvas canvas, Size size) {
  // Center of the canvas
  final center = Offset(size.width / 2, size.height / 2);

  canvas.drawCircle(center, 50, Paint()..color = Colors.blue);
}

Responsive Positioning

@override
void paint(Canvas canvas, Size size) {
  // Position at 10% from left, 20% from top
  final topLeft = Offset(size.width * 0.1, size.height * 0.2);

  // Position at 90% from left, 80% from top
  final bottomRight = Offset(size.width * 0.9, size.height * 0.8);

  canvas.drawCircle(topLeft, 20, Paint()..color = Colors.red);
  canvas.drawCircle(bottomRight, 20, Paint()..color = Colors.green);
}

Creating Rects from Offset

@override
void paint(Canvas canvas, Size size) {
  // Create a rect starting at (10, 20) with size 100x50
  final topLeft = Offset(10, 20);
  final rectSize = Size(100, 50);
  final rect = topLeft & rectSize; // Rect.fromLTWH(10, 20, 100, 50)

  canvas.drawRect(rect, Paint()..color = Colors.blue);
}
