üìö What are Canvas Drawing APIs?

Canvas Drawing APIs are methods provided by the Canvas class in Flutter (from dart:ui) that allow you to draw various shapes, lines, and graphics on the drawing surface. These are the fundamental building blocks for creating custom graphics in Flutter's CustomPainter.

Think of Canvas Drawing APIs as:

  • A toolbox of drawing methods for creating visual elements
  • Primitive shapes like lines, circles, rectangles, and arcs
  • Complex shapes like custom paths and point collections
  • The interface between your drawing logic and the rendered output
üí° Key Concept: All Canvas drawing methods require a Paint object that defines how the shape should be styled (color, stroke width, fill/stroke style, etc.). The canvas provides the drawing surface, while Paint provides the styling.

Categories of Drawing APIs

  • Primitive Shapes: Basic geometric shapes like lines, circles, rectangles, ovals, and arcs
  • Complex Shapes: Custom paths and point collections for advanced graphics
  • Text & Images: Drawing text and images on the canvas (covered separately)
  • Transformations: Rotating, scaling, and translating the canvas (covered separately)

üî∑ Primitive Shapes

drawLine(Offset p1, Offset p2, Paint paint)

Purpose: Draws a straight line between two points.

Parameters:

  • p1 - Starting point of the line
  • p2 - Ending point of the line
  • paint - Paint object defining line style (color, stroke width, etc.)

Common uses: Axes, grid lines, separators, chart lines, connectors, borders.

@override
void paint(Canvas canvas, Size size) {
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  final startPoint = Paint()..color = Colors.green.withValues(alpha: 0.7);
  final endPoint = Paint()..color = Colors.red.withValues(alpha: 0.7);

  // 1Ô∏è‚É£ Horizontal line (left ‚Üí right)
  canvas.drawLine(const Offset(40, 80), Offset(size.width - 40, 80), paint);
  canvas.drawCircle(const Offset(40, 80), 5, startPoint);
  canvas.drawCircle(Offset(size.width - 40, 80), 5, endPoint);

  // 2Ô∏è‚É£ Vertical line (top ‚Üí bottom)
  canvas.drawCircle(const Offset(80, 40), 5, startPoint);
  canvas.drawLine(const Offset(80, 40), Offset(80, size.height - 40), paint);
  canvas.drawCircle(Offset(80, size.height - 40), 5, endPoint);

  // 3Ô∏è‚É£ Slanting (diagonal) line
  canvas.drawCircle(const Offset(120, 120), 5, startPoint);
  canvas.drawLine(const Offset(120, 120), Offset(size.width - 40, size.height - 40), paint);
  canvas.drawCircle(Offset(size.width - 40, size.height - 40), 5, endPoint);
}

drawCircle(Offset center, double radius, Paint paint)

Purpose: Draws a perfect circle with a specified center and radius.

Parameters:

  • center - Center point of the circle
  • radius - Radius of the circle in logical pixels
  • paint - Paint object defining circle style

Common uses: Buttons, icons, indicators, decorative elements, progress indicators.

@override
void paint(Canvas canvas, Size size) {
  final shape1 = Offset(size.width / 2, size.height / 2 - 80);
  final shape2 = Offset(size.width / 2, size.height / 2 + 80);
  final paint1 = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.fill;
  final paint2 = Paint()
    ..color = Colors.red
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2;
  final radius = 70.0;
  canvas.drawCircle(shape1, radius, paint1);
  canvas.drawCircle(shape2, radius, paint2);
}

drawOval(Rect rect, Paint paint)

Purpose: Draws an ellipse (oval) that fits within the specified rectangle.

Parameters:

  • rect - Bounding rectangle that defines the oval's size and position
  • paint - Paint object defining oval style

Common uses: Elliptical shapes, pill-shaped buttons, decorative elements, progress indicators.

@override
void paint(Canvas canvas, Size size) {
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  // Define a rectangle that bounds the oval
  final rect = Rect.fromCenter(center: Offset(size.width / 2, size.height / 2), width: 220, height: 120);

  // Draw the rectangle that bounds the oval
  canvas.drawRect(
    rect,
    Paint()
      ..color = Colors.grey.withValues(alpha: 0.5)
      ..style = PaintingStyle.stroke
      ..strokeWidth = 1.4,
  );
  // Draw the oval inside the rectangle
  canvas.drawOval(rect, paint);
}

drawRect(Rect rect, Paint paint)

Purpose: Draws a rectangle with sharp corners.

Parameters:

  • rect - Rectangle defining position and size
  • paint - Paint object defining rectangle style

Common uses: Backgrounds, borders, grids, bars, charts, UI elements.

@override
void paint(Canvas canvas, Size size) {
  final strokePaint = Paint()
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  // 1Ô∏è‚É£ Rect.fromLTWH - Top Left (Blue)
  final rectFromLTWH = Rect.fromLTWH(50, 50, 120, 100);
  canvas.drawRect(rectFromLTWH, strokePaint..color = Colors.blue);

  // 2Ô∏è‚É£ Rect.fromPoints - Top Right (Green)
  final point1 = Offset(size.width - 200, 70);
  final point2 = Offset(size.width - 50, 170);
  final rectFromPoints = Rect.fromPoints(point1, point2);
  // Draw corner markers
  canvas.drawCircle(point1, 5, Paint()..color = Colors.grey.withValues(alpha: 0.7));
  canvas.drawCircle(point2, 5, Paint()..color = Colors.grey.withValues(alpha: 0.7));
  canvas.drawRect(rectFromPoints, strokePaint..color = Colors.green);

  // 3Ô∏è‚É£ Rect.fromLTRB - Center Left (Purple)
  final rectFromLTRB = Rect.fromLTRB(50, size.height / 2 - 60, 170, size.height / 2 + 60);
  canvas.drawRect(rectFromLTRB, strokePaint..color = Colors.purple);

  // 4Ô∏è‚É£ Rect.fromCenter - Center Right (Orange)
  final rectFromCenter = Rect.fromCenter(center: Offset(size.width - 150, size.height / 2), width: 150, height: 100);
  canvas.drawRect(rectFromCenter, strokePaint..color = Colors.orange);

  // 5Ô∏è‚É£ Rect.fromCircle - Bottom Center (Red)
  final circleCenter = Offset(size.width / 2, size.height - 150);
  final rectFromCircle = Rect.fromCircle(center: circleCenter, radius: 80);
  // Draw the circle inside the bounding box
  canvas.drawCircle(circleCenter, 80, Paint()..color = Colors.grey.withValues(alpha: 0.7));
  canvas.drawRect(rectFromCircle, strokePaint..color = Colors.red);
}

Visual guide showing how drawRect() draws rectangles using different Rect constructors: fromLTWH (blue), fromPoints (green), fromLTRB (purple), fromCenter (orange), and fromCircle (red).

drawRRect(RRect rrect, Paint paint)

Purpose: Draws a rounded rectangle with specified corner radius.

Parameters:

  • rrect - Rounded rectangle (RRect) with corner radius information
  • paint - Paint object defining rounded rectangle style

Common uses: Cards, buttons, modern UI surfaces, containers, badges.

@override
void paint(Canvas canvas, Size size) {
  final strokePaint = Paint()
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  final fillPaint = Paint()..style = PaintingStyle.fill;

  // 1Ô∏è‚É£ RRect.fromRectAndRadius - Top Left (Blue)
  final rect1 = Rect.fromLTWH(50, 50, 120, 100);
  final rrect1 = RRect.fromRectAndRadius(rect1, const Radius.circular(20));
  canvas.drawRRect(rrect1, strokePaint..color = Colors.blue);

  // 2Ô∏è‚É£ RRect.fromRectAndCorners - Top Right (Green)
  final rect2 = Rect.fromLTWH(size.width - 170, 50, 120, 100);
  final rrect2 = RRect.fromRectAndCorners(
    rect2,
    topLeft: const Radius.circular(30),
    topRight: const Radius.circular(10),
    bottomLeft: const Radius.circular(10),
    bottomRight: const Radius.circular(30),
  );
  canvas.drawRRect(rrect2, strokePaint..color = Colors.green);

  // 3Ô∏è‚É£ RRect.fromLTRBR - Center Left (Purple)
  final rrect3 = RRect.fromLTRBR(50, size.height / 2 - 60, 170, size.height / 2 + 60, const Radius.circular(25));
  canvas.drawRRect(rrect3, strokePaint..color = Colors.purple);

  // 4Ô∏è‚É£ RRect.fromLTRBXY - Center Right (Orange)
  final rrect4 = RRect.fromLTRBXY(
    size.width - 170,
    size.height / 2 - 60,
    size.width - 50,
    size.height / 2 + 60,
    15,
    30,
  );
  canvas.drawRRect(rrect4, strokePaint..color = Colors.orange);

  // 5Ô∏è‚É£ RRect.fromLTRBAndCorners - Bottom Left (Red)
  final rrect5 = RRect.fromLTRBAndCorners(
    50,
    size.height - 150,
    200,
    size.height - 50,
    topLeft: const Radius.circular(40),
    topRight: const Radius.circular(5),
    bottomLeft: const Radius.circular(5),
    bottomRight: const Radius.circular(40),
  );
  canvas.drawRRect(rrect5, strokePaint..color = Colors.red);

  // 6Ô∏è‚É£ RRect with fill - Bottom Right (Teal)
  final rect6 = Rect.fromLTWH(size.width - 200, size.height - 150, 150, 100);
  final rrect6 = RRect.fromRectAndRadius(rect6, const Radius.circular(20));
  canvas.drawRRect(rrect6, fillPaint..color = Colors.teal.withValues(alpha: 0.5));
  canvas.drawRRect(rrect6, strokePaint..color = Colors.teal);
}

drawDRRect(RRect outer, RRect inner, Paint paint)

Purpose: Draws a double rounded rectangle (donut shape) by drawing an outer rounded rectangle and cutting out an inner rounded rectangle.

Parameters:

  • outer - Outer rounded rectangle
  • inner - Inner rounded rectangle (hole)
  • paint - Paint object defining the shape style

Common uses: Donut shapes, rings, progress indicators, frames, borders with holes.

@override
void paint(Canvas canvas, Size size) {
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.fill;

  // Outer rounded rectangle
  final outerRect = Rect.fromLTWH(50, 50, 200, 150);
  final outerRRect = RRect.fromRectAndRadius(
    outerRect,
    const Radius.circular(20),
  );

  // Inner rounded rectangle (smaller, centered)
  final innerRect = Rect.fromLTWH(90, 80, 120, 90);
  final innerRRect = RRect.fromRectAndRadius(
    innerRect,
    const Radius.circular(15),
  );

  // Draw the double rounded rectangle (donut shape)
  canvas.drawDRRect(outerRRect, innerRRect, paint);
}

Visual guide showing how drawDRRect() creates a double rounded rectangle (donut shape). The outer blue rounded rectangle has a smaller inner rounded rectangle cut out from its center, creating a frame-like appearance. Both rectangles have rounded corners with different radii.

drawArc(Rect rect, double startAngle, double sweepAngle, bool useCenter, Paint paint)

Purpose: Draws an arc (portion of an ellipse) or a pie segment.

Parameters:

  • rect - Bounding rectangle that defines the ellipse
  • startAngle - Starting angle in radians (0 is at 3 o'clock)
  • sweepAngle - Angle to sweep in radians (positive = clockwise)
  • useCenter - If true, draws a pie segment; if false, draws just the arc
  • paint - Paint object defining arc style

Common uses: Pie charts, progress indicators, circular gauges, decorative arcs.

@override
void paint(Canvas canvas, Size size) {
  final strokePaint = Paint()
    ..style = PaintingStyle.stroke
    ..strokeWidth = 4;

  final fillPaint = Paint()..style = PaintingStyle.fill;

  // 1Ô∏è‚É£ Arc - Top Left (Blue) - Quarter circle, no center
  final rect1 = Rect.fromLTWH(50, 50, 120, 120);
  canvas.drawRect(
    rect1,
    Paint()
      ..color = Colors.grey.withValues(alpha: 0.5)
      ..style = PaintingStyle.stroke
      ..strokeWidth = 1.4,
  );
  canvas.drawCircle(rect1.center, 5, strokePaint..color = Colors.grey.withValues(alpha: 0.5));
  canvas.drawArc(rect1, 0, math.pi / 2, false, strokePaint..color = Colors.blue);

  // 2Ô∏è‚É£ Arc - Top Right (Green) - Half circle, with center
  final rect2 = Rect.fromLTWH(size.width - 170, 50, 120, 120);
  canvas.drawArc(rect2, 0, math.pi, true, strokePaint..color = Colors.green);

  // 3Ô∏è‚É£ Arc - Center Left (Purple) - Three-quarter circle, no center
  final rect3 = Rect.fromLTWH(50, size.height / 2 - 80, 120, 120);
  canvas.drawArc(rect3, math.pi / 4, 3 * math.pi / 2, false, strokePaint..color = Colors.purple);

  // 4Ô∏è‚É£ Arc - Center Right (Orange) - Full circle with center (pie slice)
  final rect4 = Rect.fromLTWH(size.width - 170, size.height / 2 - 80, 120, 120);
  canvas.drawArc(rect4, math.pi / 6, 2 * math.pi / 3, true, fillPaint..color = Colors.orange.withValues(alpha: 0.7));
  canvas.drawArc(rect4, math.pi / 6, 2 * math.pi / 3, true, strokePaint..color = Colors.orange);

  // 5Ô∏è‚É£ Arc - Bottom Left (Red) - Small arc, no center
  final rect5 = Rect.fromLTWH(50, size.height - 170, 120, 120);
  canvas.drawArc(rect5, math.pi / 3, math.pi / 4, false, strokePaint..color = Colors.red);

  // 6Ô∏è‚É£ Arc - Bottom Right (Teal) - Multiple arcs showing different angles
  final rect6 = Rect.fromLTWH(size.width - 170, size.height - 170, 120, 120);
  // Draw multiple arcs with different start angles
  canvas.drawArc(rect6, 0, math.pi / 3, false, strokePaint..color = Colors.teal);
  canvas.drawArc(rect6, math.pi / 2, math.pi / 3, false, strokePaint..color = Colors.teal.withValues(alpha: 0.7));
  canvas.drawArc(rect6, math.pi, math.pi / 3, false, strokePaint..color = Colors.teal.withValues(alpha: 0.5));
  canvas.drawArc(rect6, 3 * math.pi / 2, math.pi / 3, false, strokePaint..color = Colors.teal.withValues(alpha: 0.3));
}
