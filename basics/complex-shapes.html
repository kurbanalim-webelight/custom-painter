drawPath(Path path, Paint paint)

Purpose: Draws a custom shape defined by a Path object.

Parameters:

  • path - Path object defining the custom shape
  • paint - Paint object defining path style

Common uses: Custom polygons, curves, wave shapes, freehand drawing, signatures, complex graphics.

@override
void paint(Canvas canvas, Size size) {
  final strokePaint = Paint()
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  final fillPaint = Paint()..style = PaintingStyle.fill;

  // 1Ô∏è‚É£ Path - Top Left (Blue) - Simple line path
  final path1 = Path()
    ..moveTo(50, 50)
    ..lineTo(150, 50)
    ..lineTo(150, 150)
    ..lineTo(50, 150)
    ..close();
  canvas.drawPath(path1, strokePaint..color = Colors.blue);

  // 2Ô∏è‚É£ Path - Top Right (Green) - Triangle with fill
  final path2 = Path()
    ..moveTo(size.width - 100, 50)
    ..lineTo(size.width - 50, 150)
    ..lineTo(size.width - 150, 150)
    ..close();
  canvas.drawPath(path2, fillPaint..color = Colors.green.withValues(alpha: 0.5));
  canvas.drawPath(path2, strokePaint..color = Colors.green);

  // 3Ô∏è‚É£ Path - Center Left (Purple) - Quadratic Bezier curve
  final path3 = Path()
    ..moveTo(50, size.height / 2 - 50)
    ..quadraticBezierTo(100, size.height / 2 - 100, 150, size.height / 2 - 50);
  canvas.drawPath(path3, strokePaint..color = Colors.purple);

  // 4Ô∏è‚É£ Path - Center Right (Orange) - Cubic Bezier curve
  final path4 = Path()
    ..moveTo(size.width - 200, size.height / 2 - 50)
    ..cubicTo(
      size.width - 150,
      size.height / 2 - 100,
      size.width - 100,
      size.height / 2 + 50,
      size.width - 50,
      size.height / 2 - 50,
    );
  canvas.drawPath(path4, strokePaint..color = Colors.orange);

  // 5Ô∏è‚É£ Path - Bottom Right (Teal) - Complex path with arc
  final path6 = Path()
    ..moveTo(size.width - 200, size.height - 200)
    ..lineTo(size.width - 100, size.height - 200)
    ..arcToPoint(Offset(size.width - 100, size.height - 100), radius: const Radius.circular(50), clockwise: false)
    ..lineTo(size.width - 200, size.height - 100)
    ..close();
  canvas.drawPath(path6, fillPaint..color = Colors.teal.withValues(alpha: 0.5));
  canvas.drawPath(path6, strokePaint..color = Colors.teal);
}

drawPoints(PointMode pointMode, List<Offset> points, Paint paint)

Canvas.drawPoints is used to draw multiple point-based shapes efficiently in a single draw call. Instead of drawing each dot or line segment individually, Flutter batches them together, making this method ideal for markers, graphs, paths, and visual indicators.

canvas.drawPoints(PointMode pointMode, List<Offset> points, Paint paint);

What each parameter means

1Ô∏è‚É£ PointMode pointMode

Defines how Flutter should interpret the list of points.

Mode Meaning
PointMode.points Draws each Offset as an individual dot
PointMode.lines Draws independent line segments using pairs of points
PointMode.polygon Draws a connected polyline through all points

Important behavior:

  • PointMode.points ‚Üí every offset is rendered as a dot
  • PointMode.lines ‚Üí every two offsets form one line segment
  • PointMode.polygon ‚Üí all offsets are connected in sequence

2Ô∏è‚É£ List<Offset> points

A list of coordinates in canvas space. How these points are rendered depends entirely on the selected PointMode.

3Ô∏è‚É£ Paint paint

Controls the visual appearance of the points or lines.

‚ö†Ô∏è Important: PaintingStyle.stroke is required. PaintingStyle.fill is ignored by drawPoints.

üìå Example

@override
void paint(Canvas canvas, Size size) {
  // üî¥ POINTS MODE (individual dots)
  final pointPaint = Paint()
    ..color = Colors.red
    ..strokeWidth = 10
    ..strokeCap = StrokeCap.round
    ..style = PaintingStyle.stroke;

  final dotPoints = [
    const Offset(60, 60),
    const Offset(100, 80),
    const Offset(140, 60),
    const Offset(180, 90),
  ];

  canvas.drawPoints(ui.PointMode.points, dotPoints, pointPaint);

  // üîµ LINES MODE (independent segments)
  final linePaint = Paint()
    ..color = Colors.blue
    ..strokeWidth = 4
    ..style = PaintingStyle.stroke;

  final linePoints = [
    const Offset(40, 150),
    const Offset(120, 150),

    const Offset(160, 130),
    const Offset(220, 180),
  ];

  canvas.drawPoints(ui.PointMode.lines, linePoints, linePaint);

  // üü¢ POLYGON MODE (connected polyline)
  final polygonPaint = Paint()
    ..color = Colors.green
    ..strokeWidth = 3
    ..style = PaintingStyle.stroke;

  final polygonPoints = [
    Offset(80, 260),
    Offset(140, 230),
    Offset(200, 260),
    Offset(170, 300),
    Offset(110, 300),
  ];

  canvas.drawPoints(ui.PointMode.polygon, polygonPoints, polygonPaint);
}

drawParagraph(Paragraph paragraph, Offset offset)

Purpose: Draws text on the canvas using a Paragraph object.

Parameters:

  • paragraph - Paragraph object containing the text and styling information
  • offset - Position where the text should be drawn (top-left of the text)

Common uses: Custom text rendering, labels, annotations, dynamic text overlays, formatted text.

@override
void paint(Canvas canvas, Size size) {
  final padding = 20.0;
  final spacing = 100.0;

  // Example 1: Using TextPainter (simpler approach) - Top Left
  final textPainter1 = TextPainter(
    text: const TextSpan(
      text: '1. TextPainter\nSimple text rendering',
      style: TextStyle(color: Colors.blue, fontSize: 22, fontWeight: FontWeight.bold),
    ),
    textDirection: TextDirection.ltr,
  );
  textPainter1.layout();
  textPainter1.paint(canvas, Offset(padding, padding));

  // Example 2: Using ParagraphBuilder with single style - Top Right
  final paragraphBuilder2 = ui.ParagraphBuilder(
    ui.ParagraphStyle(textDirection: ui.TextDirection.ltr, fontSize: 20, fontWeight: FontWeight.w600),
  );
  paragraphBuilder2.pushStyle(ui.TextStyle(color: ui.Color(Colors.green.shade700.toARGB32())));
  paragraphBuilder2.addText('2. ParagraphBuilder\nSingle style text');
  final paragraph2 = paragraphBuilder2.build();
  paragraph2.layout(ui.ParagraphConstraints(width: size.width / 2 - padding * 2));
  canvas.drawParagraph(paragraph2, Offset(size.width / 2 + padding, padding));

  // Example 3: Using ParagraphBuilder with multiple styles - Center Left
  final paragraphBuilder3 = ui.ParagraphBuilder(
    ui.ParagraphStyle(textDirection: ui.TextDirection.ltr, fontSize: 20),
  );
  paragraphBuilder3.pushStyle(ui.TextStyle(color: ui.Color(Colors.purple.toARGB32()), fontWeight: FontWeight.bold));
  paragraphBuilder3.addText('3. Multi-style: ');
  paragraphBuilder3.pop();
  paragraphBuilder3.pushStyle(ui.TextStyle(color: ui.Color(Colors.orange.toARGB32()), fontWeight: FontWeight.bold));
  paragraphBuilder3.addText('Bold ');
  paragraphBuilder3.pop();
  paragraphBuilder3.pushStyle(ui.TextStyle(color: ui.Color(Colors.teal.toARGB32()), fontStyle: FontStyle.italic));
  paragraphBuilder3.addText('Italic ');
  paragraphBuilder3.pop();
  paragraphBuilder3.pushStyle(
    ui.TextStyle(color: ui.Color(Colors.red.toARGB32()), decoration: TextDecoration.underline),
  );
  paragraphBuilder3.addText('Underline');
  final paragraph3 = paragraphBuilder3.build();
  paragraph3.layout(ui.ParagraphConstraints(width: size.width / 2 - padding * 2));
  canvas.drawParagraph(paragraph3, Offset(padding, size.height / 2 - 60));

  // Example 4: Using TextPainter with alignment - Center Right
  final textPainter4 = TextPainter(
    text: const TextSpan(
      text: '4. Centered Text\nwith TextPainter',
      style: TextStyle(color: Colors.orange, fontSize: 22),
    ),
    textDirection: TextDirection.ltr,
    textAlign: TextAlign.center,
  );
  textPainter4.layout(maxWidth: size.width / 2 - padding * 2);
  textPainter4.paint(canvas, Offset(size.width / 2 + padding, size.height / 2 - 60));

  // Example 5: Using ParagraphBuilder with RTL text direction - Bottom Left
  final paragraphBuilder5 = ui.ParagraphBuilder(
    ui.ParagraphStyle(textDirection: ui.TextDirection.rtl, fontSize: 20),
  );
  paragraphBuilder5.pushStyle(ui.TextStyle(color: ui.Color(Colors.red.toARGB32())));
  paragraphBuilder5.addText('5. RTL Text Direction');
  final paragraph5 = paragraphBuilder5.build();
  paragraph5.layout(ui.ParagraphConstraints(width: size.width / 2 - padding * 2));
  canvas.drawParagraph(paragraph5, Offset(padding, size.height - spacing));

  // Example 6: Using TextPainter with background - Bottom Right
  final textPainter6 = TextPainter(
    text: const TextSpan(
      text: '6. Text with\nBackground',
      style: TextStyle(color: Colors.white, fontSize: 20, fontWeight: FontWeight.bold),
    ),
    textDirection: TextDirection.ltr,
  );
  textPainter6.layout();
  final textRect = Rect.fromLTWH(
    size.width / 2 + padding,
    size.height - spacing - 10,
    textPainter6.width + 30,
    textPainter6.height + 30,
  );
  canvas.drawRect(textRect, Paint()..color = Colors.teal);
  textPainter6.paint(canvas, Offset(size.width / 2 + padding + 15, size.height - spacing + 5));
}

üé® Comprehensive Example: Cityscape at Sunset

This example demonstrates how to combine multiple Canvas drawing methods to create a complete scene. The cityscape uses various drawing APIs including gradients, shapes, paths, points, and text rendering.

@override
void paint(Canvas canvas, Size size) {
  // Sky gradient background using Rect
  final skyGradient = Paint()
    ..shader = ui.Gradient.linear(
      Offset(0, 0),
      Offset(0, size.height * 0.6),
      [Colors.deepOrange.shade300, Colors.purple.shade300, Colors.blue.shade900],
      [0.0, 0.5, 1.0],
    );
  canvas.drawRect(Rect.fromLTWH(0, 0, size.width, size.height * 0.6), skyGradient);

  // Ground using Rect
  final groundPaint = Paint()..color = Colors.grey.shade800;
  canvas.drawRect(Rect.fromLTWH(0, size.height * 0.6, size.width, size.height * 0.4), groundPaint);

  // Sun using Circle
  final sunCenter = Offset(size.width * 0.8, size.height * 0.2);
  final sunPaint = Paint()..color = Colors.yellow;
  canvas.drawCircle(sunCenter, 60, sunPaint);

  // Sun rays using Arc
  final rayPaint = Paint()
    ..color = Colors.orange
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;
  for (int i = 0; i < 8; i++) {
    final angle = (i * 2 * math.pi / 8) - math.pi / 2;
    final rayRect = Rect.fromCircle(center: sunCenter, radius: 80);
    canvas.drawArc(rayRect, angle, math.pi / 8, false, rayPaint);
  }

  // Stars using Points
  final starPoints = <Offset>[];
  final starPaint = Paint()
    ..color = Colors.white
    ..strokeWidth = 2
    ..strokeCap = StrokeCap.round;
  for (int i = 0; i < 20; i++) {
    starPoints.add(Offset((i * 37.5) % size.width, (i * 23.7) % (size.height * 0.4)));
  }
  canvas.drawPoints(ui.PointMode.points, starPoints, starPaint);

  // Buildings using Rect
  final buildingPaint = Paint()..color = Colors.grey.shade900;
  final buildings = [
    Rect.fromLTWH(50, size.height * 0.4, 80, size.height * 0.2),
    Rect.fromLTWH(150, size.height * 0.35, 100, size.height * 0.25),
    Rect.fromLTWH(270, size.height * 0.3, 90, size.height * 0.3),
    Rect.fromLTWH(380, size.height * 0.38, 70, size.height * 0.22),
    Rect.fromLTWH(470, size.height * 0.32, 110, size.height * 0.28),
  ];
  for (final building in buildings) {
    canvas.drawRect(building, buildingPaint);
  }

  // Windows using RRect (rounded rectangles)
  final windowPaint = Paint()..color = Colors.yellow.shade700;
  for (final building in buildings) {
    final windowWidth = 15.0;
    final windowHeight = 20.0;
    final spacing = 20.0;
    for (double y = building.top + 15; y < building.bottom - 15; y += spacing + windowHeight) {
      for (double x = building.left + 10; x < building.right - 10; x += spacing + windowWidth) {
        final windowRect = RRect.fromRectAndRadius(
          Rect.fromLTWH(x, y, windowWidth, windowHeight),
          const Radius.circular(3),
        );
        canvas.drawRRect(windowRect, windowPaint);
      }
    }
  }

  // Road using Path (curved road)
  final roadPath = Path()
    ..moveTo(0, size.height * 0.7)
    ..quadraticBezierTo(size.width * 0.3, size.height * 0.65, size.width * 0.6, size.height * 0.7)
    ..quadraticBezierTo(size.width * 0.8, size.height * 0.75, size.width, size.height * 0.7);
  final roadPaint = Paint()
    ..color = Colors.grey.shade600
    ..style = PaintingStyle.stroke
    ..strokeWidth = 40;
  canvas.drawPath(roadPath, roadPaint);

  // Road center line using Line
  final linePath = Path()
    ..moveTo(-2, size.height * 0.7)
    ..quadraticBezierTo(size.width * 0.3, size.height * 0.65, size.width * 0.6, size.height * 0.7)
    ..quadraticBezierTo(size.width * 0.8, size.height * 0.75, size.width, size.height * 0.7);
  final linePaint = Paint()
    ..color = Colors.yellow
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;
  canvas.drawPath(linePath, linePaint);

  // Cloud using Oval
  final cloudPaint = Paint()..color = Colors.white.withValues(alpha: 0.8);
  final clouds = [
    Offset(size.width * 0.2, size.height * 0.15),
    Offset(size.width * 0.5, size.height * 0.1),
    Offset(size.width * 0.7, size.height * 0.18),
  ];
  for (final cloudCenter in clouds) {
    canvas.drawOval(Rect.fromCenter(center: cloudCenter, width: 80, height: 40), cloudPaint);
    canvas.drawOval(
      Rect.fromCenter(center: Offset(cloudCenter.dx - 20, cloudCenter.dy), width: 60, height: 50),
      cloudPaint,
    );
    canvas.drawOval(
      Rect.fromCenter(center: Offset(cloudCenter.dx + 20, cloudCenter.dy), width: 60, height: 50),
      cloudPaint,
    );
  }

  // Tree using Path (complex shape)
  final treePath = Path()
    ..moveTo(size.width * 0.15, size.height * 0.6)
    ..lineTo(size.width * 0.15, size.height * 0.75)
    ..moveTo(size.width * 0.1, size.height * 0.65)
    ..quadraticBezierTo(size.width * 0.15, size.height * 0.55, size.width * 0.2, size.height * 0.65)
    ..quadraticBezierTo(size.width * 0.15, size.height * 0.5, size.width * 0.1, size.height * 0.6)
    ..close();
  final treePaint = Paint()
    ..color = Colors.green.shade700
    ..style = PaintingStyle.fill;
  canvas.drawPath(treePath, treePaint);
  final trunkPaint = Paint()..color = Colors.brown.shade700;
  canvas.drawRect(Rect.fromLTWH(size.width * 0.14, size.height * 0.75, 8, 20), trunkPaint);

  // Title using Paragraph
  final paragraphBuilder = ui.ParagraphBuilder(
    ui.ParagraphStyle(textDirection: ui.TextDirection.ltr, fontSize: 28, fontWeight: FontWeight.bold),
  );
  paragraphBuilder.pushStyle(ui.TextStyle(color: ui.Color(Colors.black.toARGB32())));
  paragraphBuilder.addText('Cityscape at Sunset');
  final paragraph = paragraphBuilder.build();
  paragraph.layout(ui.ParagraphConstraints(width: size.width));
  canvas.drawParagraph(paragraph, Offset(size.width / 2 - paragraph.maxIntrinsicWidth / 2, 20));

  // Subtitle using TextPainter
  final textPainter = TextPainter(
    text: const TextSpan(
      text: 'Demonstrating all canvas drawing concepts',
      style: TextStyle(color: Colors.black, fontSize: 16),
    ),
    textDirection: TextDirection.ltr,
  );
  textPainter.layout();
  textPainter.paint(canvas, Offset(size.width / 2 - textPainter.width / 2, 60));
}

A comprehensive example demonstrating all Canvas drawing methods working together: gradients with drawRect, shapes with drawCircle and drawOval, curves with drawArc and drawPath, points with drawPoints, rounded rectangles with drawRRect, and text with drawParagraph and TextPainter.

Complex Shapes

Method Parameters Description
drawPath(Path path, Paint paint) Path path, Paint paint Draws a custom shape defined by a Path
drawPoints(PointMode pointMode, List<Offset> points, Paint paint) PointMode, List<Offset>, Paint Draws multiple points, lines, or polygons
drawRawPoints(PointMode pointMode, Float32List points, Paint paint) PointMode, Float32List, Paint Draws points from raw float array (more efficient for many points)

Other Drawing Methods

Method Parameters Description
drawImage(Image image, Offset offset, Paint paint) Image, Offset, Paint Draws an image on the canvas
drawImageRect(Image image, Rect src, Rect dst, Paint paint) Image, Rect src, Rect dst, Paint Draws a portion of an image scaled to fit a rectangle
drawImageNine(Image image, Rect center, Rect dst, Paint paint) Image, Rect center, Rect dst, Paint Draws an image using 9-patch scaling
drawParagraph(Paragraph paragraph, Offset offset) Paragraph, Offset Draws text on the canvas
drawPicture(Picture picture) Picture picture Draws a recorded picture
drawVertices(Vertices vertices, BlendMode blendMode, Paint paint) Vertices, BlendMode, Paint Draws vertices (for 3D-like effects)
drawAtlas(Image atlas, List<RSTransform> transforms, List<Rect> rects, List<Color>? colors, BlendMode? blendMode, Rect? cullRect, Paint paint) Image, List<RSTransform>, List<Rect>, List<Color>?, BlendMode?, Rect?, Paint Draws multiple sprites from an atlas efficiently
