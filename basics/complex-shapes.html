drawPath(Path path, Paint paint)

Purpose: Draws a custom shape defined by a Path object.

Parameters:

  • path - Path object defining the custom shape
  • paint - Paint object defining path style

Common uses: Custom polygons, curves, wave shapes, freehand drawing, signatures, complex graphics.

@override
void paint(Canvas canvas, Size size) {
  final strokePaint = Paint()
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  final fillPaint = Paint()..style = PaintingStyle.fill;

  // 1Ô∏è‚É£ Path - Top Left (Blue) - Simple line path
  final path1 = Path()
    ..moveTo(50, 50)
    ..lineTo(150, 50)
    ..lineTo(150, 150)
    ..lineTo(50, 150)
    ..close();
  canvas.drawPath(path1, strokePaint..color = Colors.blue);

  // 2Ô∏è‚É£ Path - Top Right (Green) - Triangle with fill
  final path2 = Path()
    ..moveTo(size.width - 100, 50)
    ..lineTo(size.width - 50, 150)
    ..lineTo(size.width - 150, 150)
    ..close();
  canvas.drawPath(path2, fillPaint..color = Colors.green.withValues(alpha: 0.5));
  canvas.drawPath(path2, strokePaint..color = Colors.green);

  // 3Ô∏è‚É£ Path - Center Left (Purple) - Quadratic Bezier curve
  final path3 = Path()
    ..moveTo(50, size.height / 2 - 50)
    ..quadraticBezierTo(100, size.height / 2 - 100, 150, size.height / 2 - 50);
  canvas.drawPath(path3, strokePaint..color = Colors.purple);

  // 4Ô∏è‚É£ Path - Center Right (Orange) - Cubic Bezier curve
  final path4 = Path()
    ..moveTo(size.width - 200, size.height / 2 - 50)
    ..cubicTo(
      size.width - 150,
      size.height / 2 - 100,
      size.width - 100,
      size.height / 2 + 50,
      size.width - 50,
      size.height / 2 - 50,
    );
  canvas.drawPath(path4, strokePaint..color = Colors.orange);

  // 5Ô∏è‚É£ Path - Bottom Right (Teal) - Complex path with arc
  final path6 = Path()
    ..moveTo(size.width - 200, size.height - 200)
    ..lineTo(size.width - 100, size.height - 200)
    ..arcToPoint(Offset(size.width - 100, size.height - 100), radius: const Radius.circular(50), clockwise: false)
    ..lineTo(size.width - 200, size.height - 100)
    ..close();
  canvas.drawPath(path6, fillPaint..color = Colors.teal.withValues(alpha: 0.5));
  canvas.drawPath(path6, strokePaint..color = Colors.teal);
}

drawPoints(PointMode pointMode, List<Offset> points, Paint paint)

Canvas.drawPoints is used to draw multiple point-based shapes efficiently in a single draw call. Instead of drawing each dot or line segment individually, Flutter batches them together, making this method ideal for markers, graphs, paths, and visual indicators.

canvas.drawPoints(PointMode pointMode, List<Offset> points, Paint paint);

What each parameter means

1Ô∏è‚É£ PointMode pointMode

Defines how Flutter should interpret the list of points.

Mode Meaning
PointMode.points Draws each Offset as an individual dot
PointMode.lines Draws independent line segments using pairs of points
PointMode.polygon Draws a connected polyline through all points

Important behavior:

  • PointMode.points ‚Üí every offset is rendered as a dot
  • PointMode.lines ‚Üí every two offsets form one line segment
  • PointMode.polygon ‚Üí all offsets are connected in sequence

2Ô∏è‚É£ List<Offset> points

A list of coordinates in canvas space. How these points are rendered depends entirely on the selected PointMode.

3Ô∏è‚É£ Paint paint

Controls the visual appearance of the points or lines.

‚ö†Ô∏è Important: PaintingStyle.stroke is required. PaintingStyle.fill is ignored by drawPoints.

üìå Example

@override
void paint(Canvas canvas, Size size) {
  // üî¥ POINTS MODE (individual dots)
  final pointPaint = Paint()
    ..color = Colors.red
    ..strokeWidth = 10
    ..strokeCap = StrokeCap.round
    ..style = PaintingStyle.stroke;

  final dotPoints = [
    const Offset(60, 60),
    const Offset(100, 80),
    const Offset(140, 60),
    const Offset(180, 90),
  ];

  canvas.drawPoints(ui.PointMode.points, dotPoints, pointPaint);

  // üîµ LINES MODE (independent segments)
  final linePaint = Paint()
    ..color = Colors.blue
    ..strokeWidth = 4
    ..style = PaintingStyle.stroke;

  final linePoints = [
    const Offset(40, 150),
    const Offset(120, 150),

    const Offset(160, 130),
    const Offset(220, 180),
  ];

  canvas.drawPoints(ui.PointMode.lines, linePoints, linePaint);

  // üü¢ POLYGON MODE (connected polyline)
  final polygonPaint = Paint()
    ..color = Colors.green
    ..strokeWidth = 3
    ..style = PaintingStyle.stroke;

  final polygonPoints = [
    Offset(80, 260),
    Offset(140, 230),
    Offset(200, 260),
    Offset(170, 300),
    Offset(110, 300),
  ];

  canvas.drawPoints(ui.PointMode.polygon, polygonPoints, polygonPaint);
}

drawParagraph(Paragraph paragraph, Offset offset)

Purpose: Draws text on the canvas using a Paragraph object built with ParagraphBuilder. Supports wrapping, styling, and precise positioning.

Parameters:

  • paragraph ‚Äî A laid-out Paragraph (you must call paragraph.layout(ParagraphConstraints(...)) before drawing).
  • offset ‚Äî Position where the text is drawn (top-left of the paragraph).

Common uses: Custom text rendering, labels, annotations, dynamic text overlays, formatted multi-line text with emojis.

class ParagraphPainter extends CustomPainter {
  @override
  void paint(Canvas canvas, Size size) {
    // 1Ô∏è‚É£ Create a ParagraphStyle
    final paragraphStyle = ui.ParagraphStyle(
      textAlign: TextAlign.left,
      fontSize: 18,
      maxLines: 3,
    );

    // 2Ô∏è‚É£ Create a TextStyle
    final textStyle = ui.TextStyle(
      color: Colors.black,
      fontWeight: FontWeight.w500,
    );

    // 3Ô∏è‚É£ Create ParagraphBuilder
    final builder = ui.ParagraphBuilder(paragraphStyle)
      ..pushStyle(textStyle)
      ..addText(
        'Hello Canvas üëã\n'
        'This text is drawn using drawParagraph.\n'
        'It supports wrapping and styling.',
      );

    // 4Ô∏è‚É£ Build the paragraph
    final paragraph = builder.build();

    // 5Ô∏è‚É£ Layout the paragraph (VERY IMPORTANT)
    paragraph.layout(
      ui.ParagraphConstraints(
        width: size.width - 40,
      ),
    );

    // 6Ô∏è‚É£ Draw it on canvas
    canvas.drawParagraph(
      paragraph,
      const Offset(20, 40),
    );
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}

Output of drawParagraph: multi-line text with emoji (üëã), wrapping based on ParagraphConstraints(width: size.width - 40), and positioning at Offset(20, 40).
