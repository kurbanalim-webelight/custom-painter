üìö What is Path?

Path is a class in Flutter (from dart:ui) that represents a sequence of connected lines and curves that form a shape. It's the foundation for drawing custom, complex shapes that can't be created with simple primitives like circles or rectangles.

Think of Path as:

  • A drawing tool that lets you create any shape by connecting points with lines and curves
  • A vector path similar to SVG paths or pen tools in design software
  • A blueprint for complex shapes that you can fill, stroke, or use for clipping
  • A sequence of drawing commands that define how to move and draw on the canvas
üí° Key Concept: Path is used with canvas.drawPath(path, paint) to render custom shapes. You build the path step by step using methods like moveTo(), lineTo(), quadraticBezierTo(), etc., then draw it once.

Common Use Cases

  • Custom shapes: Stars, polygons, arrows, badges
  • Curved designs: Waves, hearts, organic shapes
  • Complex graphics: Logos, icons, decorative elements
  • Clipping regions: Mask areas for images or other content
  • Hit testing: Check if a point is inside a custom shape
  • Animations: Morphing between shapes, path animations

üéØ Essential Path Methods

moveTo(double x, double y)

Purpose: Move the "pen" to a specific point without drawing. This sets the starting point for the next drawing operation.

Common uses: Starting a new path segment, jumping to a different location before drawing.

Think of it as:

  • A painter lifting off his hand from the paper and placing it at a new position
  • A pencil lifting off the paper and moving to a new position
@override
void paint(Canvas canvas, Size size) {
  final path = Path();

  // Move to starting point (no line drawn)
  path.moveTo(50, 50);
}

lineTo(double x, double y)

Purpose: Draw a straight line from the current point to the specified point.

Common uses: Creating polygons, connecting points, drawing straight edges.

@override
void paint(Canvas canvas, Size size) {
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2;

  // Draw a triangle
  path.moveTo(100, 50); // Top point
  canvas.drawCircle(Offset(100, 50), 5, Paint()..color = Colors.red.withAlpha(180));
  path.lineTo(50, 150); // Bottom left
  canvas.drawCircle(Offset(50, 150), 10, Paint()..color = Colors.red.withAlpha(180));
  path.lineTo(150, 150); // Bottom right
  canvas.drawCircle(Offset(150, 150), 15, Paint()..color = Colors.red.withAlpha(180));
  path.close(); // Close back to start

  canvas.drawPath(path, paint);
}

quadraticBezierTo(double x1, double y1, double x2, double y2)

Purpose: Draw a quadratic Bezier curve using one control point.

Parameters:

  • x1, y1 - Control point (influences the curve shape)
  • x2, y2 - End point

Common uses: Smooth curves, rounded corners, organic shapes.

@override
void paint(Canvas canvas, Size size) {
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2;

  // Start point (Green circle)
  path.moveTo(50, 100);
  canvas.drawCircle(Offset(50, 100), 8, Paint()
    ..color = Colors.green
    ..style = PaintingStyle.fill);
  canvas.drawCircle(Offset(50, 100), 8, Paint()
    ..color = Colors.white
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2);

  // Control point (Red circle)
  final controlPoint = Offset(100, 20);
  canvas.drawCircle(controlPoint, 8, Paint()
    ..color = Colors.red
    ..style = PaintingStyle.fill);
  canvas.drawCircle(controlPoint, 8, Paint()
    ..color = Colors.white
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2);

  // End point (Orange circle)
  final endPoint = Offset(150, 100);
  canvas.drawCircle(endPoint, 8, Paint()
    ..color = Colors.orange
    ..style = PaintingStyle.fill);
  canvas.drawCircle(endPoint, 8, Paint()
    ..color = Colors.white
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2);

  // Helper lines (grey dashed) - from start to control, control to end
  canvas.drawLine(Offset(50, 100), controlPoint, Paint()
    ..color = Colors.grey.withAlpha(128)
    ..strokeWidth = 1);
  canvas.drawLine(controlPoint, endPoint, Paint()
    ..color = Colors.grey.withAlpha(128)
    ..strokeWidth = 1);

  // Quadratic Bezier curve
  path.quadraticBezierTo(controlPoint.dx, controlPoint.dy, endPoint.dx, endPoint.dy);

  canvas.drawPath(path, paint);
}

Each point has a white border for visibility. cubicTo(double x1, double y1, double x2, double y2, double x3, double y3)

Purpose: Draw a cubic Bezier curve using two control points for more complex curves.

Parameters:

  • x1, y1 - First control point
  • x2, y2 - Second control point
  • x3, y3 - End point

Common uses: Complex curves, S-shapes, wave patterns, smooth transitions.

@override
void paint(Canvas canvas, Size size) {
  // Define points for cubic Bezier curve
  const startPoint = Offset(50, 150);
  const controlPoint1 = Offset(100, 20); // First control point
  const controlPoint2 = Offset(200, 300); // Second control point
  const endPoint = Offset(250, 150);

  // Draw the cubic Bezier curve
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  path.moveTo(startPoint.dx, startPoint.dy);
  // cubicTo takes: control1.x, control1.y, control2.x, control2.y, end.x, end.y
  path.cubicTo(controlPoint1.dx, controlPoint1.dy, controlPoint2.dx, controlPoint2.dy, endPoint.dx, endPoint.dy);
  canvas.drawPath(path, paint);

  // Draw reference lines (dashed) to show control points' influence
  final dashedLinePaint = Paint()
    ..color = Colors.grey.withValues(alpha: 0.5)
    ..style = PaintingStyle.stroke
    ..strokeWidth = 1;

  // Line from start to first control point
  final helperPath1 = Path()
    ..moveTo(startPoint.dx, startPoint.dy)
    ..lineTo(controlPoint1.dx, controlPoint1.dy);
  canvas.drawPath(helperPath1, dashedLinePaint);

  // Line from first control point to second control point
  final helperPath2 = Path()
    ..moveTo(controlPoint1.dx, controlPoint1.dy)
    ..lineTo(controlPoint2.dx, controlPoint2.dy);
  canvas.drawPath(helperPath2, dashedLinePaint);

  // Line from second control point to end point
  final helperPath3 = Path()
    ..moveTo(controlPoint2.dx, controlPoint2.dy)
    ..lineTo(endPoint.dx, endPoint.dy);
  canvas.drawPath(helperPath3, dashedLinePaint);

  // Draw reference points with different colors
  // Start point (green)
  final startPointPaint = Paint()
    ..color = Colors.green
    ..style = PaintingStyle.fill;
  canvas.drawCircle(startPoint, 6, startPointPaint);
  canvas.drawCircle(
    startPoint,
    6,
    Paint()
      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );

  // First control point (red)
  final controlPoint1Paint = Paint()
    ..color = Colors.red
    ..style = PaintingStyle.fill;
  canvas.drawCircle(controlPoint1, 6, controlPoint1Paint);
  canvas.drawCircle(
    controlPoint1,
    6,
    Paint()
      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );

  // Second control point (purple)
  final controlPoint2Paint = Paint()
    ..color = Colors.purple
    ..style = PaintingStyle.fill;
  canvas.drawCircle(controlPoint2, 6, controlPoint2Paint);
  canvas.drawCircle(
    controlPoint2,

      ..color = Colors.white
      ..style = PaintingStyle.stroke

  );

  // End point (orange)

    ..color = Colors.orange

    endPoint,

      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );
}

addRect(Rect rect)

Purpose: Add a rectangle to the path as a closed contour.

Common uses: Combining rectangles with other shapes, creating complex paths from simple shapes.

@override
void paint(Canvas canvas, Size size) {
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.fill;

  // Add a rectangle to the path
  path.addRect(Rect.fromLTWH(50, 50, 100, 80));

  // Add another rectangle
  path.addRect(Rect.fromLTWH(200, 50, 100, 80));

  canvas.drawPath(path, paint);
}

addOval(Rect rect)

Purpose: Add an oval (ellipse) to the path. The rect defines the bounding box of the oval.

Common uses: Adding circles and ellipses to paths, creating rounded shapes.

@override
void paint(Canvas canvas, Size size) {
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.fill;

  // Add an oval (circle when width == height)
  path.addOval(Rect.fromLTWH(50, 50, 100, 100));

  // Add an ellipse
  path.addOval(Rect.fromLTWH(200, 50, 150, 80));

  canvas.drawPath(path, paint);
}

close()

Purpose: Close the current contour by drawing a line from the current point back to the starting point.

Common uses: Closing polygons, creating filled shapes, ensuring paths are complete.

@override
void paint(Canvas canvas, Size size) {
  // Define points for a triangle/polygon
  const startPoint = Offset(150, 50); // Top point
  const point2 = Offset(50, 200); // Bottom left
  const point3 = Offset(250, 200); // Bottom right

  // Create path and draw lines
  final path = Path();
  final paint = Paint()
    ..color = Colors.blue
    ..style = PaintingStyle.stroke
    ..strokeWidth = 3;

  path.moveTo(startPoint.dx, startPoint.dy);
  path.lineTo(point2.dx, point2.dy);
  path.lineTo(point3.dx, point3.dy);
  path.close(); // Closes the path by drawing a line back to startPoint

  canvas.drawPath(path, paint);

  // Draw reference lines to show the path structure
  final dashedLinePaint = Paint()
    ..color = Colors.grey.withValues(alpha: 0.5)
    ..style = PaintingStyle.stroke
    ..strokeWidth = 1;

  // Line from start to point2
  final line1 = Path()
    ..moveTo(startPoint.dx, startPoint.dy)
    ..lineTo(point2.dx, point2.dy);
  canvas.drawPath(line1, dashedLinePaint);

  // Line from point2 to point3
  final line2 = Path()
    ..moveTo(point2.dx, point2.dy)
    ..lineTo(point3.dx, point3.dy);
  canvas.drawPath(line2, dashedLinePaint);

  // Highlight the closing line (from point3 back to start) with a different color
  final closingLinePaint = Paint()
    ..color = Colors.red
    ..style = PaintingStyle.stroke
    ..strokeWidth = 2;
  final closingLine = Path()
    ..moveTo(point3.dx, point3.dy)
    ..lineTo(startPoint.dx, startPoint.dy);
  canvas.drawPath(closingLine, closingLinePaint);

  // Draw reference points at all vertices
  // Start point (green) - also the closing point
  final startPointPaint = Paint()
    ..color = Colors.green
    ..style = PaintingStyle.fill;
  canvas.drawCircle(startPoint, 6, startPointPaint);
  canvas.drawCircle(
    startPoint,
    6,
    Paint()
      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );

  // Point 2 (red)
  final point2Paint = Paint()
    ..color = Colors.red
    ..style = PaintingStyle.fill;
  canvas.drawCircle(point2, 6, point2Paint);
  canvas.drawCircle(
    point2,
    6,
    Paint()
      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );

  // Point 3 (purple)
  final point3Paint = Paint()
    ..color = Colors.purple
    ..style = PaintingStyle.fill;
  canvas.drawCircle(point3, 6, point3Paint);
  canvas.drawCircle(
    point3,
    6,
    Paint()
      ..color = Colors.white
      ..style = PaintingStyle.stroke
      ..strokeWidth = 2,
  );
}

üîß Other Important Methods

relativeLineTo(double dx, double dy)

Purpose: Draw a line relative to the current point (adds dx, dy to current position).

final path = Path();
path.moveTo(50, 50);
path.relativeLineTo(100, 0); // Move 100 right from current point
path.relativeLineTo(0, 50); // Move 50 down from current point

addRRect(RRect rrect)

Purpose: Add a rounded rectangle to the path.

final path = Path();
path.addRRect(RRect.fromRectAndRadius(
  Rect.fromLTWH(50, 50, 100, 80),
  Radius.circular(10),
));

addPolygon(List<Offset> points, bool close)

Purpose: Add a polygon from a list of points. If close is true, connects the last point to the first.

final path = Path();
path.addPolygon([
  Offset(50, 50),
  Offset(150, 50),
  Offset(150, 150),
  Offset(50, 150),
], true); // true = close the polygon

addPath(Path path, Offset offset)

Purpose: Add another path to this path, optionally offset by the given offset.

final path1 = Path();
path1.addRect(Rect.fromLTWH(0, 0, 50, 50));

final path2 = Path();
path2.addPath(path1, Offset(100, 100)); // Add path1 at offset

reset()

Purpose: Clear all path data, resetting it to an empty path.

final path = Path();
path.moveTo(50, 50);
path.lineTo(100, 100);
path.reset(); // Path is now empty

shift(Offset offset)

Purpose: Move the entire path by the given offset.

final path = Path();
path.addRect(Rect.fromLTWH(0, 0, 50, 50));
path.shift(Offset(100, 100)); // Move path 100 right, 100 down

contains(Offset point)

Purpose: Check if a point is inside the path (hit testing).

final path = Path();
path.addRect(Rect.fromLTWH(50, 50, 100, 100));

final point = Offset(75, 75);
if (path.contains(point)) {
  // Point is inside the path
}

arcTo(Rect rect, double startAngle, double sweepAngle, bool forceMoveTo)

Purpose: Add an arc to the path. The arc is part of an oval defined by the rect.

final path = Path();
path.arcTo(
  Rect.fromLTWH(50, 50, 100, 100),
  0, // Start angle in radians
  3.14159, // Sweep angle (half circle)
  false, // Don't force moveTo
);

arcToPoint(Offset endRadius, {Radius radius, double rotation, bool largeArc, bool clockwise})

Purpose: Add an arc from the current point to a target point.

final path = Path();
path.moveTo(50, 100);
path.arcToPoint(
  Offset(150, 100),
  radius: Radius.circular(50),
  largeArc: false,
  clockwise: true,
);
